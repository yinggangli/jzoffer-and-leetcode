## 快速排序
`代码`<br>
```python
def quickSort(a, left, right):
    if left > right:
        return
    
    i = left
    j = right
    temp = a[left]

    while i != j:
        while a[j] >= temp and i < j:
            j -= 1
        while a[i] <= temp and i < j:
            i += 1
        
        t = a[j]
        a[j] = a[i]
        a[i] = t
    
    a[left] = a[i]
    a[i] = temp
    quickSort(a, left, i - 1)
    quickSort(a, i + 1, right)
```
